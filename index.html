  <button onclick="enviar()">Enviar</button>

  

  <p id="msg"></p>

  

  <script>

    let estado = "";

    function setEstado(e) {

      estado = e;

      document.getElementById("estadoElegido").textContent = e;

    }

    

    async function enviar() {

      const usuario = document.getElementById("usuario").value;

      const comentario = document.getElementById("comentario").value;

      

      if (!estado || !usuario) {

        document.getElementById("msg").textContent = "⚠️ Selecciona un estado y escribe tu correo.";

        return;

      }

      

      try {

        const response = await fetch("AQUI_VA_TU_URL_WEBAPP", {

          method: "POST",

          body: JSON.stringify({ usuario, estado, comentario }),

          headers: { "Content-Type": "application/json" }

        });

        

        const result = await response.json();

        if (result.result === "success") {

          document.getElementById("msg").textContent = "✅ Registro enviado con éxito.";

        } else {

          document.getElementById("msg").textContent = "❌ Error al enviar.";

        }

      } catch (err) {

        document.getElementById("msg").textContent = "❌ Error de conexión.";

      }

    }

  </script>
